```typescript
// TypeScript 基本数据类型示例

// 1. 基础类型
let isDone: boolean = false;
let age: number = 25;
let firstName: string = "张三";
let symbol: symbol = Symbol("id");
let bigIntValue: bigint = 100n;

// 2. 数组类型
let numbers: number[] = [1, 2, 3, 4, 5];
let fruits: Array<string> = ["苹果", "香蕉", "橘子"];

// 3. 元组类型 - 固定长度和类型的数组
let position: [number, number] = [10, 20];
let userInfo: [string, number, boolean] = ["张三", 25, true];

// 4. 枚举类型
enum Color {
    Red,
    Green,
    Blue
}
let favoriteColor: Color = Color.Blue;

// 5. any 类型 - 可以是任意类型
let anything: any = 42;
anything = "现在是字符串";
anything = true;

// 6. unknown 类型 - 类型安全的 any
let value: unknown = 4;
// value.toFixed(); // 错误：需要类型检查
if (typeof value === "number") {
    value.toFixed(); // 正确：经过类型检查
}

// 7. void 类型 - 通常用于函数无返回值
function logMessage(message: string): void {
    console.log(message);
}

// 8. null 和 undefined
let nullValue: null = null;
let undefinedValue: undefined = undefined;

// 9. never 类型 - 永远不会返回的函数
function throwError(message: string): never {
    throw new Error(message);
}

// 10. object 类型
let user: object = {
    name: "张三",
    age: 25
};
```
