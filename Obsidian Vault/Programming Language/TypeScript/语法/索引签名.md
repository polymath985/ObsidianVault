```typescript
// 索引签名的基本语法
interface StringDictionary {
    [key: string]: string;  // 字符串索引签名
}

const dict: StringDictionary = {
    name: "张三",
    city: "北京",
    job: "工程师"
};

// 数字索引签名
interface NumberArray {
    [index: number]: string;
}

const fruits: NumberArray = ["苹果", "香蕉", "橘子"];
console.log(fruits[0]); // "苹果"
```

```typescript
// 混合索引签名示例
interface MixedIndex {
    [key: string]: string | number;  // 字符串索引
    [index: number]: string;         // 数字索引
    
    // 注意：数字索引的返回类型必须是字符串索引返回类型的子类型
}

const mixed: MixedIndex = {
    0: "第一项",      // 数字索引
    1: "第二项",      // 数字索引
    name: "示例",     // 字符串索引
    count: 100       // 字符串索引
};
```

```typescript
// 只读索引签名
interface ReadonlyStringDictionary {
    readonly [key: string]: string;
}

const config: ReadonlyStringDictionary = {
    apiUrl: "https://api.example.com",
    version: "1.0.0"
```
